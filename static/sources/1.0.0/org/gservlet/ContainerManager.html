69   * 
70   * Manages the registration and the reloading of
71   * a servlet, filter or listener into the web container

78  	/**
79  	 * The servlet context object
80  	 */
81  	protected final ServletContext context;
82  	/**
83  	 * The map of dynamic invocation handlers
84  	 */
85  	protected final Map<String, DynamicInvocationHandler> handlers = new HashMap<>();
86  	/**
87  	 * The script manager object
88  	 */
89  	protected ScriptManager scriptManager;
90  	/**
91  	 * The logger object
92  	 */
93  	protected final Logger logger = Logger.getLogger(getClass().getName());

95  	/**
96  	 * 
97  	 * Constructs a ContainerManager for the given servlet context
98  	 * 
99  	 * @param context the servlet context
100 	 * @throws ServletException the ServletException
101 	 * 
102 	 */
103 	public ContainerManager(ServletContext context) throws ServletException {
104 		try {
105 			this.context = context;
106 			context.setAttribute(HANDLERS, handlers);
107 		} catch (Exception e) {
108 			throw new ServletException(e);
109 		}
110 	}

113 	/**
114 	 * 
115 	 * Initializes the application for the given directory
116 	 *
117 	 * @param directory the scripts parent directory
118 	 * @param listeners the ScriptListener list
119 	 * @throws ServletException the ServletException
120 	 * @throws ScriptException  the ScriptException
121 	 * 
122 	 */
123 	public void init(String directory, List<ScriptListener> listeners) throws ServletException, ScriptException {
124 		File folder = new File(directory + File.separator + SCRIPTS_FOLDER);
125 		scriptManager = new ScriptManager(folder);
126 		addScriptListeners(listeners);
127 		loadScripts(folder);
128 	}
129 
130 	/**
131 	 * 
132 	 * Loads and registers the servlets, filters or listeners
133 	 * 
134 	 * @param folder the scripts folder
135 	 * @throws ServletException the ServletException
136 	 * @throws ScriptException  the ScriptException
137 	 * 
138 	 */
139 	protected void loadScripts(File folder) throws ServletException, ScriptException {
140 		if (folder.exists()) {
141 			File[] files = folder.listFiles();
142 			if (files != null) {
143 				for (File file : files) {
144 					if (file.isFile()) {
145 						register(scriptManager.loadObject(file));
146 					} else {
147 						loadScripts(file);
148 					}
149 				}
150 			}
151 			watch(folder);
152 		}
153 	}

155 	/**
156 	 * 
157 	 * Registers a servlet, filter or listener into the web container
158 	 * 
159 	 * @param object the object
160 	 * @throws ServletException the ServletException
161 	 * 
162 	 */
163 	public void register(Object object) throws ServletException {
164 		Annotation[] annotations = object.getClass().getAnnotations();
165 		for (Annotation annotation : annotations) {
166 			if (annotation instanceof Servlet) {
167 				addServlet(object);
168 			} else if (annotation instanceof Filter) {
169 				addFilter(object);
170 			} else if (annotation instanceof ContextListener || annotation instanceof ContextAttributeListener
171 					|| annotation instanceof RequestListener || annotation instanceof RequestAttributeListener
172 					|| annotation instanceof SessionListener || annotation instanceof SessionAttributeListener
173 					|| annotation instanceof SessionActivationListener || annotation instanceof SessionBindingListener
174 					|| annotation instanceof SessionIdListener) {
175 				addListener(object);
176 			}
177 		}
178 	}

180 	/**
181 	 * 
182 	 * Registers a servlet into the web container
183 	 * 
184 	 * @param object the servlet object
185 	 * @throws ServletException the ServletException
186 	 * 
187 	 */
188 	protected void addServlet(Object object) throws ServletException {
189 		Servlet annotation = object.getClass().getAnnotation(Servlet.class);
190 		String name = object.getClass().getName();
191 		ServletRegistration registration = context.getServletRegistration(name);
192 		if (registration == null) {
193 			DynamicInvocationHandler handler = new DynamicInvocationHandler(object);
194 			Object servlet = Proxy.newProxyInstance(this.getClass().getClassLoader(),
195 					new Class[] { javax.servlet.Servlet.class }, handler);
196 			handlers.put(name, handler);
197 			ServletRegistration.Dynamic dynamic = context.addServlet(name, (javax.servlet.Servlet) servlet);
198 			dynamic.setLoadOnStartup(annotation.loadOnStartup());
199 			dynamic.setAsyncSupported(annotation.asyncSupported());
200 			if (annotation.value().length > 0) {
201 				dynamic.addMapping(annotation.value());
202 			}
203 			if (annotation.urlPatterns().length > 0) {
204 				dynamic.addMapping(annotation.urlPatterns());
205 			}
206 			for (InitParam param : annotation.initParams()) {
207 				dynamic.setInitParameter(param.name(), param.value());
208 			}
209 			MultipartConfig multiPartConfig = object.getClass().getAnnotation(MultipartConfig.class);
210 			if(multiPartConfig != null) {
211 				dynamic.setMultipartConfig(new MultipartConfigElement(multiPartConfig));
212 			}
213 			ServletSecurity servletSecurity = object.getClass().getAnnotation(ServletSecurity.class);
214 			if(servletSecurity != null) {
215 				dynamic.setServletSecurity(new ServletSecurityElement(servletSecurity));
216 			}
217 		} else {
218 			String message = "The servlet with the name " + name
219 					+ " has already been registered. Please use a different name or package";
220 			throw new ServletException(message);
221 		}
222 	}

224 	/**
225 	 * 
226 	 * Registers a filter into the web container
227 	 * 
228 	 * @param object  the filter object
229 	 * @throws ServletException the ServletException
230 	 * 
231 	 */
232 	protected void addFilter(Object object) throws ServletException {
233 		Filter annotation = object.getClass().getAnnotation(Filter.class);
234 		String name = object.getClass().getName();
235 		FilterRegistration registration = context.getFilterRegistration(name);
236 		if (registration == null) {
237 			DynamicInvocationHandler handler = new DynamicInvocationHandler(object);
238 			Object filter = Proxy.newProxyInstance(this.getClass().getClassLoader(),
239 					new Class[] { javax.servlet.Filter.class }, handler);
240 			handlers.put(name, handler);
241 			FilterRegistration.Dynamic dynamic = context.addFilter(name, (javax.servlet.Filter) filter);
242 		    dynamic.setAsyncSupported(annotation.asyncSupported());
243 			Collection<DispatcherType> dispatcherTypes = Arrays.asList(annotation.dispatcherTypes());
244 			if (annotation.value().length > 0) {
245 				dynamic.addMappingForUrlPatterns(EnumSet.copyOf(dispatcherTypes), true, annotation.value());
246 			}
247 			if (annotation.urlPatterns().length > 0) {
248 				dynamic.addMappingForUrlPatterns(EnumSet.copyOf(dispatcherTypes), true, annotation.urlPatterns());
249 			}
250 			for (InitParam param : annotation.initParams()) {
251 				dynamic.setInitParameter(param.name(), param.value());
252 			}
253 		} else {
254 			String message = "The filter with the name " + name
255 					+ " has already been registered. Please use a different name or package";
256 			throw new ServletException(message);
257 		}
258 	}

260 	/**
261 	 * 
262 	 * Registers a listener into the web container
263 	 * 
264 	 * @param object  the listener object
265 	 * 
266 	 */
267 	protected void addListener(Object object) {
268 		DynamicInvocationHandler handler = new DynamicInvocationHandler(object);
269 		EventListener listener = null;
270 		if (object instanceof ServletContextAttributeListener) {
271 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
272 					new Class[] { ServletContextAttributeListener.class }, handler);
273 		} else if (object instanceof ServletRequestListener) {
274 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
275 					new Class[] { ServletRequestListener.class }, handler);
276 		} else if (object instanceof ServletRequestAttributeListener) {
277 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
278 					new Class[] { ServletRequestAttributeListener.class }, handler);
279 		} else if (object instanceof HttpSessionListener) {
280 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
281 					new Class[] { HttpSessionListener.class }, handler);
282 		} else if (object instanceof HttpSessionAttributeListener) {
283 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
284 					new Class[] { HttpSessionAttributeListener.class }, handler);
285 		} else if (object instanceof HttpSessionActivationListener) {
286 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
287 					new Class[] { HttpSessionActivationListener.class }, handler);
288 		} else if (object instanceof HttpSessionBindingListener) {
289 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
290 					new Class[] { HttpSessionBindingListener.class }, handler);
291 		} else if (object instanceof HttpSessionIdListener) {
292 			listener = (EventListener) Proxy.newProxyInstance(this.getClass().getClassLoader(),
293 					new Class[] { HttpSessionIdListener.class }, handler);
294 		}
295 		if (listener != null) {
296 			context.addListener(listener);
297 		} else if (object instanceof AbstractContextListener) {
298 			AbstractContextListener contextListener = (AbstractContextListener) object;
299 			contextListener.contextInitialized(new ServletContextEvent(context));
300 		}
301 		handlers.put(object.getClass().getName(), handler);
302 	}

304 	/**
305 	 * 
306 	 * Watches the scripts folder for file changes
307 	 * 
308 	 * @param folder the scripts folder
309 	 * 
310 	 */
311 	protected void watch(File folder) {
312 		new FileWatcher(folder).addListener(new FileAdapter() {
313 			
314 			@Override
315 			public void onCreated(FileEvent event) {
316 				File file = event.getFile();
317 				if (file.isFile()) {
318 					logger.info("processing script " + file.getName());
319 					process(file);
320 				}
321 			}
322 			
323 			@Override
324 			public void onModified(FileEvent event) {
325 				onCreated(event);
326 			}
327 			
328 
329 		}).watch();
330 	}

332 	/**
333 	 * 
334 	 * Processes a script file
335 	 * 
336 	 * @param script the script file
337 	 * 
338 	 */
339 	protected void process(File script) {
340 		try {
341 			Object object = scriptManager.loadObject(script);
342 			Annotation[] annotations = object.getClass().getAnnotations();
343 			for (Annotation annotation : annotations) {
344 				if (annotation instanceof RequestListener || annotation instanceof ContextAttributeListener
345 						|| annotation instanceof RequestAttributeListener || annotation instanceof SessionListener
346 						|| annotation instanceof SessionAttributeListener) {
347 					reload(object);
348 				} else if(annotation instanceof Servlet) {
349 					reloadServlet((AbstractServlet) object);
350 					
351 				} else if(annotation instanceof Filter) {
352 					reloadFilter((AbstractFilter) object);
353 				}
354 			}
355 		} catch (Exception e) {
356 			logger.log(Level.INFO, "exception during reload", e);
357 		}
358 	}

360 	/**
361 	 * 
362 	 * Reloads a servlet, filter or a listener into the web container
363 	 * 
364 	 * @param object the object to be reloaded
365 	 * 
366 	 */
367 	protected void reload(Object object) {
368 		DynamicInvocationHandler handler = handlers.get(object.getClass().getName());
369 		if (handler != null) {
370 			handler.setTarget(object);
371 		}
372 	}
373 	
374 	/**
375 	 * 
376 	 * Reloads a servlet
377 	 * 
378 	 * @param servlet the servlet to be reloaded
379 	 * @throws ServletException the ServletException
380 	 * 
381 	 */
382 	protected void reloadServlet(AbstractServlet servlet) throws ServletException {
383 		reload(servlet);
384 		DefaultServletConfig config = new DefaultServletConfig();
385 		config.setServletContext(context);
386 		Servlet annotation = servlet.getClass().getAnnotation(Servlet.class);
387 		for (InitParam param : annotation.initParams()) {
388 			config.addInitParameter(param.name(), param.value());
389 		}
390 		servlet.init(config);
391 	}

394 	 * 
395 	 * Reloads a filter
396 	 * 
397 	 * @param filter the filter to be reloaded
398 	 * @throws ServletException the ServletException
399 	 * 
400 	 */
401 	protected void reloadFilter(AbstractFilter filter) throws ServletException {
402 		reload(filter);
403 		DefaultFilterConfig config = new DefaultFilterConfig();
404 		config.setServletContext(context);
405 		Filter annotation = filter.getClass().getAnnotation(Filter.class);
406 		for (InitParam param : annotation.initParams()) {
407 			config.addInitParameter(param.name(), param.value());
408 		}
409 		filter.init(config);
410 	}

412 	/**
413 	 * 
414 	 * Calls the destroy method of the ServletContextListener objects when the
415 	 * ServletContext is about to be shutdown
416 	 * 
417 	 */
418 	public void shutDown() {
419 		for (DynamicInvocationHandler handler : handlers.values()) {
420 			Object target = handler.getTarget();
421 			if (target instanceof AbstractContextListener) {
422 				AbstractContextListener contextListener = (AbstractContextListener) target;
423 				contextListener.contextDestroyed(new ServletContextEvent(context));
424 			}
425 		}
426 		handlers.clear();
427 	}
