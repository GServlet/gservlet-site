20  package org.gservlet;
21  
22  import static org.gservlet.Constants.APP_CONFIG_FILE;
23  import java.io.File;
24  import java.io.FileReader;
25  import java.io.IOException;
26  import java.nio.file.WatchService;
27  import java.util.Properties;
28  import java.util.logging.Level;
29  import java.util.logging.Logger;
30  import javax.servlet.ServletContext;
31  import javax.servlet.ServletContextEvent;
32  import javax.servlet.ServletContextListener;
33  import javax.servlet.annotation.WebListener;
34  
35  /**
36   * 
37   * Bootstraps the application and starts the registration of the servlets, filters,
38   * listeners into the web container.
39   * 
40   * @author Mamadou Lamine Ba
41   * 
42   */
43  @WebListener
44  public class StartupListener implements ServletContextListener {
45  
46  	/**
47  	 * The container initializer object
48  	 */
49  	protected ContainerInitializer initializer;
50  	/**
51  	 * The database manager object
52  	 */
53  	protected DatabaseManager databaseManager;
54  	/**
55  	 * The logger object
56  	 */
57  	protected final Logger logger = Logger.getLogger(getClass().getName());
58  
59  	/**
60  	 * 
61  	 * initializes the application and the registration process 
62  	 * 
63  	 * @param event the ServletContextEvent containing the ServletContext that is
64  	 *              being initialized
65  	 * 
66  	 */
67  	@Override
68  	public void contextInitialized(ServletContextEvent event) {
69  		try {
70  			ServletContext context = event.getServletContext();
71  			initializer = new ContainerInitializer(context);
72  			databaseManager = new DatabaseManager(context);
73  			File root = new File(context.getRealPath("/"));
74  			File configuration = new File(root + File.separator + APP_CONFIG_FILE);
75  			if(configuration.exists()) {
76  				databaseManager.setupDataSource(loadConfiguration(configuration));
77  			}
78  			watch(root);
79  			logger.info("application started on context " + context.getContextPath());
80  		} catch (Exception e) {
81  			logger.log(Level.INFO, "exception during contextInitialized method", e);
82  		}
83  	}
84  	
85  	/**
86  	 * Loads the configuration file properties
87  	 * 
88  	 * @param file the configuration file
89  	 * @throws IOException throws an Exception if the configuration file is invalid
90  	 * @return the properties of the configuration file
91  	 */
92  	public Properties loadConfiguration(File file) throws IOException {
93  		Properties configuration = new Properties();
94  		try(FileReader reader = new FileReader(file)) {
95  			configuration.load(reader);
96  		}
97  		return configuration;
98  	}
99  	
100 	/**
101 	 * Watches the configuration folder for file changes
102 	 *
103 	 * @param folder the configuration folder
104 	 */
105 	protected void watch(File folder) {
106 		new FileWatcher(folder).addListener(new FileAdapter() {
107 			@Override
108 			public void onCreated(FileEvent event) {
109 				File file = event.getFile();
110 				if (file.getName().equals(APP_CONFIG_FILE)) {
111 					logger.info("processing configuration file " + file.getName());
112 					try {
113 						databaseManager.setupDataSource(loadConfiguration(file));
114 					} catch (IOException e) {
115 						logger.log(Level.INFO, "exception during reload", e);
116 					}
117 				}
118 			}
119 		}).watch();
120 	}
121 
122 
123 	/**
124 	 * 
125 	 * Shutdowns the application in a clean way
126 	 * 
127 	 * @param event the ServletContextEvent containing the ServletContext that is
128 	 *              being destroyed
129 	 * 
130 	 */
131 	@Override
132 	public void contextDestroyed(ServletContextEvent event) {
133 		initializer.destroy();
134 		databaseManager.destroy();
135 		for (WatchService watchService : FileWatcher.getWatchServices()) {
136 			try {
137 				watchService.close();
138 			} catch (IOException e) {
139 				// the exception is ignored
140 			}
141 		}
142 	}
143 
144 	/**
145 	 * 
146 	 * Returns the ContainerInitializer object
147 	 * 
148 	 * @return the ContainerInitializer object
149 	 * 
150 	 */
151 	public ContainerInitializer getInitializer() {
152 		return initializer;
153 	}
154 
155 	/**
156 	 * 
157 	 * Returns the DatabaseManager object
158 	 * 
159 	 * @return the DatabaseManager object
160 	 * 
161 	 */
162 	public DatabaseManager getDatabaseManager() {
163 		return databaseManager;
164 	}
165 
166 }
