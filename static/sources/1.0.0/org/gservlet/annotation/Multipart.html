20  package org.gservlet.annotation;
21  
22  import java.lang.annotation.ElementType;
23  import java.lang.annotation.Retention;
24  import java.lang.annotation.RetentionPolicy;
25  import java.lang.annotation.Target;
26  
27  /**
28   * Annotation that may be specified on a servlet class, indicating that instances of the
29   * <tt>Servlet</tt> expect requests that conform to the <tt>multipart/form-data</tt> MIME type.
30   *
31   * @author Mamadou Lamine Ba
32   */
33  @Target(ElementType.TYPE)
34  @Retention(RetentionPolicy.RUNTIME)
35  public @interface Multipart {
36  
37      /**
38       * The directory location where files will be stored
39       *
40       * @return the directory location where files will be stored
41       */
42      String location() default "";
43  
44      /**
45       * The maximum size allowed for uploaded files.
46       * 
47       * <p>
48       * The default is <tt>-1L</tt>, which means unlimited.
49       *
50       * @return the maximum size allowed for uploaded files
51       */
52      long maxFileSize() default -1L;
53  
54      /**
55       * The maximum size allowed for <tt>multipart/form-data</tt> requests
56       * 
57       * <p>
58       * The default is <tt>-1L</tt>, which means unlimited.
59       *
60       * @return the maximum size allowed for <tt>multipart/form-data</tt> requests
61       */
62      long maxRequestSize() default -1L;
63  
64      /**
65       * The size threshold after which the file will be written to disk
66       *
67       * @return the size threshold after which the file will be written to disk
68       */
69      int fileSizeThreshold() default 0;
70  }
