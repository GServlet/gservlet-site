20  package org.gservlet;
21  
22  import java.io.Serializable;
23  import java.util.Enumeration;
24  import javax.servlet.ServletContext;
25  import javax.servlet.http.HttpSession;
26  
27  /**
28   * 
29   * A wrapper class around the HttpSession class
30   * 
31   * @author Mamadou Lamine Ba
32   * 
33   */
34  @SuppressWarnings("serial")
35  public class SessionWrapper implements HttpSession, Serializable {
36  
37  	/**
38  	 * The session object
39  	 */
40  	protected final transient HttpSession session;
41  	
42  	/**
43  	* 
44  	* Constructs a SessionWrapper for the given HttpSession
45  	* 
46  	* @param session the HttpSession object 
47  	*  
48  	*/
49  	public SessionWrapper(HttpSession session) {
50  		this.session = session;
51  	}
52  
53  	/**
54  	 * Returns the object bound with the specified name in this session, or
55  	 * <code>null</code> if no object is bound under the name.
56  	 *
57  	 * @param name a string specifying the name of the object
58  	 *
59  	 * @return the object with the specified name
60  	 *
61  	 * @exception IllegalStateException if this method is called on an invalidated session
62  	 * 
63  	 */
64  	@Override
65  	public Object getAttribute(String name) {
66  		return session.getAttribute(name);
67  	}
68  
69  	/**
70  	 * Returns an <code>Enumeration</code> of <code>String</code> objects containing
71  	 * the names of all the objects bound to this session.
72  	 *
73  	 * @return an <code>Enumeration</code> of <code>String</code> objects specifying
74  	 *         the names of all the objects bound to this session
75  	 *
76  	 * @exception IllegalStateException if this method is called on an invalidated session
77  	 * 
78  	 */
79  	@Override
80  	public Enumeration<String> getAttributeNames() {
81  		return session.getAttributeNames();
82  	}
83  
84  	/**
85  	 *
86  	 * Returns the time when this session was created, measured in milliseconds
87  	 * since midnight January 1, 1970 GMT.
88  	 *
89  	 * @return a <code>long</code> specifying when this session was created,
90  	 *         expressed in milliseconds since 1/1/1970 GMT
91  	 *
92  	 * @exception IllegalStateException if this method is called on an invalidated session
93  	 */
94  
95  	@Override
96  	public long getCreationTime() {
97  		return session.getCreationTime();
98  	}
99  
100 	/**
101 	 * Returns a string containing the unique identifier assigned to this session.
102 	 * The identifier is assigned by the servlet container and is implementation
103 	 * dependent.
104 	 * 
105 	 * @return a string specifying the identifier assigned to this session
106 	 */
107 	@Override
108 	public String getId() {
109 		return session.getId();
110 	}
111 
112 	/**
113 	 *
114 	 * Returns the last time the client sent a request associated with this session,
115 	 * as the number of milliseconds since midnight January 1, 1970 GMT, and marked
116 	 * by the time the container received the request.
117 	 *
118 	 * <p>
119 	 * Actions that your application takes, such as getting or setting a value
120 	 * associated with the session, do not affect the access time.
121 	 *
122 	 * @return a <code>long</code> representing the last time the client sent a
123 	 *         request associated with this session, expressed in milliseconds since
124 	 *         1/1/1970 GMT
125 	 *
126 	 * @exception IllegalStateException if this method is called on an invalidated session
127 	 * 
128 	 */
129 	@Override
130 	public long getLastAccessedTime() {
131 		return session.getLastAccessedTime();
132 	}
133 
134 	/**
135 	 * Returns the maximum time interval, in seconds, that the servlet container
136 	 * will keep this session open between client accesses. After this interval, the
137 	 * servlet container will invalidate the session. The maximum time interval can
138 	 * be set with the <code>setMaxInactiveInterval</code> method.
139 	 *
140 	 * <p>
141 	 * A return value of zero or less indicates that the session will never timeout.
142 	 *
143 	 * @return an integer specifying the number of seconds this session remains open
144 	 *         between client requests
145 	 *
146 	 * @see #setMaxInactiveInterval
147 	 */
148 	@Override
149 	public int getMaxInactiveInterval() {
150 		return session.getMaxInactiveInterval();
151 	}
152 
153 	/**
154 	 * Returns the ServletContext to which this session belongs.
155 	 * 
156 	 * @return The ServletContext object for the web application
157 	 * @since Servlet 2.3
158 	 */
159 	@Override
160 	public ServletContext getServletContext() {
161 		return session.getServletContext();
162 	}
163 
164 	/**
165 	 *
166 	 * @deprecated As of Version 2.1, this method is deprecated and has no
167 	 *             replacement. It will be removed in a future version of the Java
168 	 *             Servlet API.
169 	 *
170 	 * @return the @see javax.servlet.http.HttpSessionContext for this session.
171 	 */
172 	@Deprecated
173 	public javax.servlet.http.HttpSessionContext getSessionContext() {
174 		return session.getSessionContext();
175 	}
176 
177 	/**
178 	 * @deprecated As of Version 2.2, this method is replaced by
179 	 *             {@link #getAttribute}.
180 	 *
181 	 * @param name a string specifying the name of the object
182 	 *
183 	 * @return the object with the specified name
184 	 *
185 	 * @exception IllegalStateException if this method is called on an invalidated
186 	 *                                  session
187 	 */
188 	@Deprecated
189 	public Object getValue(String name) {
190 		return session.getValue(name);
191 	}
192 
193 	/**
194 	 * @deprecated As of Version 2.2, this method is replaced by
195 	 *             {@link #getAttributeNames}
196 	 *
197 	 * @return an array of <code>String</code> objects specifying the names of all
198 	 *         the objects bound to this session
199 	 *
200 	 * @exception IllegalStateException if this method is called on an invalidated
201 	 *                                  session
202 	 */
203 	@Deprecated
204 	public String[] getValueNames() {
205 		return session.getValueNames();
206 	}
207 
208 	/**
209 	 * Invalidates this session then unbinds any objects bound to it.
210 	 *
211 	 * @exception IllegalStateException if this method is called on an already
212 	 *                                  invalidated session
213 	 */
214 	@Override
215 	public void invalidate() {
216 		session.invalidate();
217 	}
218 
219 	/**
220 	 * Returns <code>true</code> if the client does not yet know about the session
221 	 * or if the client chooses not to join the session. For example, if the server
222 	 * used only cookie-based sessions, and the client had disabled the use of
223 	 * cookies, then a session would be new on each request.
224 	 *
225 	 * @return <code>true</code> if the server has created a session, but the client
226 	 *         has not yet joined
227 	 *
228 	 * @exception IllegalStateException if this method is called on an already
229 	 *                                  invalidated session
230 	 */
231 	@Override
232 	public boolean isNew() {
233 		return session.isNew();
234 	}
235 
236 	/**
237 	 * @deprecated As of Version 2.2, this method is replaced by
238 	 *             {@link #setAttribute}
239 	 *
240 	 * @param name  the name to which the object is bound; cannot be null
241 	 *
242 	 * @param value the object to be bound; cannot be null
243 	 *
244 	 * @exception IllegalStateException if this method is called on an invalidated
245 	 *                                  session
246 	 */
247 	@Deprecated
248 	public void putValue(String name, Object value) {
249 		session.putValue(name, value);
250 	}
251 
252 	/**
253 	 * Removes the object bound with the specified name from this session. If the
254 	 * session does not have an object bound with the specified name, this method
255 	 * does nothing.
256 	 *
257 	 * <p>
258 	 * After this method executes, and if the object implements
259 	 * <code>HttpSessionBindingListener</code>, the container calls
260 	 * <code>HttpSessionBindingListener.valueUnbound</code>. The container then
261 	 * notifies any <code>HttpSessionAttributeListener</code>s in the web
262 	 * application.
263 	 *
264 	 * @param name the name of the object to remove from this session
265 	 *
266 	 * @exception IllegalStateException if this method is called on an invalidated
267 	 *                                  session
268 	 */
269 	@Override
270 	public void removeAttribute(String name) {
271 		session.removeAttribute(name);
272 	}
273 
274 	/**
275 	 * @deprecated As of Version 2.2, this method is replaced by
276 	 *             {@link #removeAttribute}
277 	 *
278 	 * @param name the name of the object to remove from this session
279 	 *
280 	 * @exception IllegalStateException if this method is called on an invalidated
281 	 *                                  session
282 	 */
283 	@Deprecated
284 	public void removeValue(String name) {
285 		session.removeValue(name);
286 	}
287 
288 	/**
289 	 * Binds an object to this session, using the name specified. If an object of
290 	 * the same name is already bound to the session, the object is replaced.
291 	 *
292 	 * <p>
293 	 * After this method executes, and if the new object implements
294 	 * <code>HttpSessionBindingListener</code>, the container calls
295 	 * <code>HttpSessionBindingListener.valueBound</code>. The container then
296 	 * notifies any <code>HttpSessionAttributeListener</code>s in the web
297 	 * application.
298 	 * 
299 	 * <p>
300 	 * If an object was already bound to this session of this name that implements
301 	 * <code>HttpSessionBindingListener</code>, its
302 	 * <code>HttpSessionBindingListener.valueUnbound</code> method is called.
303 	 *
304 	 * <p>
305 	 * If the value passed in is null, this has the same effect as calling
306 	 * <code>removeAttribute()</code>.
307 	 *
308 	 *
309 	 * @param name  the name to which the object is bound; cannot be null
310 	 *
311 	 * @param value the object to be bound
312 	 *
313 	 * @exception IllegalStateException if this method is called on an invalidated
314 	 *                                  session
315 	 */
316 	@Override
317 	public void setAttribute(String name, Object value) {
318 		Serializable serializable = (Serializable) value;
319 		session.setAttribute(name, serializable);
320 	}
321 
322 	/**
323 	 * Specifies the time, in seconds, between client requests before the servlet
324 	 * container will invalidate this session.
325 	 *
326 	 * <p>
327 	 * An <tt>interval</tt> value of zero or less indicates that the session should
328 	 * never timeout.
329 	 *
330 	 * @param interval An integer specifying the number of seconds
331 	 */
332 	@Override
333 	public void setMaxInactiveInterval(int interval) {
334 		session.setMaxInactiveInterval(interval);
335 	}
336 
337 	/**
338 	* 
339 	* Sets an attribute
340 	* 
341 	* @param name the attribute name
342 	* @param value the attribute value
343 	* 
344 	*/
345 	public void propertyMissing(String name, Object value) {
346 		setAttribute(name, value);
347 	}
348 
349 	/**
350 	* 
351 	* Gets an attribute value
352 	* 
353 	* @param name the attribute name
354 	* @return the attribute value
355 	*/
356 	public Object propertyMissing(String name) {
357 		return getAttribute(name);
358 	}
