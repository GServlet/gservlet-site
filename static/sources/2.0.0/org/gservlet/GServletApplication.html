92  		this.context = context;
93  		this.realPath = context.getRealPath("/");
94  	}

96  	/**
97  	 * Starts the initialization process
98  	 *
99  	 */
100 	public void start() {
101 		try {
102 			initContainerManager();
103 			initDatabaseManager();
104 			logger.info("application started on context " + context.getContextPath());
105 		} catch (Exception e) {
106 			logger.log(Level.SEVERE, "exception during contextInitialized method", e);
107 		}
108 	}
109 
110 	/**
111 	 * Creates and initializes the container manager
112 	 *
113 	 *
114 	 * @throws ServletException the ServletException
115 	 * @throws ScriptException  the  ScriptException
116 	 * 
117 	 */
118 	private void initContainerManager() throws ServletException, ScriptException {
119 		containerManager = new ContainerManager(context);
120 		containerManager.init(realPath, listeners);
121 	}
122 	
123 	/**
124 	 * Creates and initializes the database manager
125 	 *
126 	 * @throws IOException the exception
127 	 * 
128 	 */
129 	private void initDatabaseManager() throws IOException {
130 		databaseManager = new DatabaseManager(context);
131 		File root = new File(realPath);
132 		databaseManager.setupDataSource(loadConfiguration(new File(root + File.separator + APP_CONFIG_FILE)));
133 		watch(root);
134 	}
135 
136 	/**
137 	 * Starts the initialization process on a SpringBoot application
138 	 *
139 	 */
140 	public void startOnSpringBoot() {
141 		realPath = System.getProperty("user.dir") + File.separator + "src/main/resources";
142 		if (new File(realPath).exists()) {
143 			start();
144 		} else {
145 			String folder = "BOOT-INF/classes/";
146 			explodeClassPathResources(context.getRealPath("/"), folder);
147 			realPath = context.getRealPath("/") + folder;
148 			start();
149 		}
150 		context.addListener(new DefaultRequestListener());
151 	}
152 
153 	/**
154 	 * Loads the configuration file properties
155 	 * 
156 	 * @param file the configuration file
157 	 * @throws IOException throws an Exception if the configuration file is invalid
158 	 * @return the properties of the configuration file
159 	 */
160 	private Properties loadConfiguration(File file) throws IOException {
161 		Properties configuration = new Properties();
162 		if(file.exists()) {
163 			try (FileReader reader = new FileReader(file)) {
164 				configuration.load(reader);
165 			}
166 		}
167 		return configuration;
168 	}
169 
170 	/**
171 	 * Watches the root folder for a change of the configuration file
172 	 *
173 	 * @param folder the root folder
174 	 */
175 	protected void watch(File folder) {
176 		new FileWatcher(folder).addFileListener(new FileAdapter() {
177 			
178 			@Override
179 			public void onCreated(FileEvent event) {
180 				File file = event.getFile();
181 				if (file.getName().equals(APP_CONFIG_FILE)) {
182 					logger.info("processing configuration file " + file.getName());
183 					try {
184 						databaseManager.setupDataSource(loadConfiguration(file));
185 					} catch (IOException e) {
186 						logger.log(Level.INFO, "exception during reload", e);
187 					}
188 				}
189 			}
190 
191 			@Override
192 			public void onModified(FileEvent event) {
193 				onCreated(event);
194 			}
195 			
196 		}).watch();
197 	}

199 	/**
200 	 * 
201 	 * Stops the application and frees the resources
202 	 *  
203 	 */
204 	public void stop() {
205 		containerManager.shutDown();
206 		databaseManager.shutDown();
207 		for (WatchService watchService : FileWatcher.getWatchServices()) {
208 			try {
209 				watchService.close();
210 			} catch (IOException e) {
211 				// the exception is ignored
212 			}
213 		}
214 	}

238 	/**
239 	 * 
240 	 * Explodes the classpath resources
241 	 * 
242 	 * @param root  the root folder
243 	 * 
244 	 * @param folder the resources folder
245 	 * 
246 	 * 
247 	 */
248 	private void explodeClassPathResources(String root, String folder) {
249 		try {
250 			List<Path> paths = getClassPathResources(folder);
251 			for (Path resourcePath : paths) {
252 				String resource = resourcePath.toString();
253 				if (resource.startsWith("/")) {
254 					resource = resource.substring(1, resource.length());
255 				}
256 				if (resource.indexOf(".") != -1)
257 					try (InputStream inputStream = getClass().getClassLoader().getResourceAsStream(resource)) {
258 						File file = new File(root + File.separator + resource);
259 						if (!file.getParentFile().exists()) {
260 							file.getParentFile().mkdirs();
261 						}
262 						Files.copy(inputStream, file.toPath());
263 					}
264 			}
265 
266 		} catch (IOException e) {
267 			logger.log(Level.SEVERE, "exception encountered while exploding the classpath resources", e);
268 		}
269 	}
270 
271 	/**
272 	 * 
273 	 * Returns the list of classpath resources
274 	 * 
275 	 * @param folder the base folder
276 	 * 
277 	 * @return the list of classpath resources
278 	 * 
279 	 */
280 	private List<Path> getClassPathResources(String folder) throws IOException {
281 		List<Path> paths = new ArrayList<>();
282 		String jarPath = getClass().getProtectionDomain().getCodeSource().getLocation().getPath();
283 		URI uri = URI.create("jar:" + jarPath);
284 		try (FileSystem fs = FileSystems.newFileSystem(uri, new HashMap<>())) {
285 			try(Stream<Path> stream = Files.walk(fs.getPath(folder + Constants.SCRIPTS_FOLDER))) {
286 				paths = stream.collect(Collectors.toList());
287 			}
288 			paths.add(fs.getPath(folder + Constants.APP_CONFIG_FILE));
289 		}
290 		return paths;
291 	}

305 	/**
306 	 *  Registers a new ScriptListener
307 	 * 
308 	 * @param listener the ScriptListener
309 	 * 
310 	 */
311 	public void addScriptListener(ScriptListener listener) {
312 		if(containerManager != null) {
313 			containerManager.getScriptManager().addScriptListener(listener);
314 		} else {
315 			listeners.add(listener);
316 		}
317 	}
318 	
319 	/**
320 	 *  Registers a ScriptListener list
321 	 * 
322 	 * @param listeners the ScriptListener list
323 	 * 
324 	 */
325 	public void addScriptListeners(List<ScriptListener> listeners) {
326 		if(containerManager != null) {
327 			containerManager.getScriptManager().addScriptListeners(listeners);
328 		} else {
329 			this.listeners.addAll(listeners);
330 		}
331 	}
332 	
333 	/**
334 	 * 
335 	 * Returns the list of script listeners
336 	 * 
337 	 * @return the list of script listeners
338 	 */
339 	public List<ScriptListener> getScriptListeners() {
340 		if(containerManager != null) {
341 			return containerManager.getScriptManager().getScriptListeners();
342 		} else {
343 			return listeners;
344 		}
345 	}
